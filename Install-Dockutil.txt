#!/bin/bash

# Variables
pkgURL="https://<yourstorage>.blob.core.windows.net/<container>/dockutil.pkg"  # ‚Üê Replace this
tempPkg="/tmp/dockutil.pkg"

# Check if dockutil is already installed
if [ ! -f "/usr/local/bin/dockutil" ]; then
    echo "Downloading dockutil.pkg from Azure..."
    curl -L "$pkgURL" -o "$tempPkg"

    echo "Installing dockutil..."
    sudo installer -pkg "$tempPkg" -target /

    echo "Cleaning up..."
    rm -f "$tempPkg"

    echo "dockutil installed successfully."
else
    echo "dockutil already installed."
fi

exit 0
